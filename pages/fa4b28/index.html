			LEFT JOIN question_detail b ON b.question_id = a.question_id
		GROUP BY a.device_id, b.difficult_level
	) b
	ON b.device_id = a.device_id
WHERE a.university = '浙江大学'
GROUP BY b.difficult_level
order by 	correct_rate
